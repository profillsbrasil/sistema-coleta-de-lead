function mockVisualViewport(height: number) {
  const listeners: Record<string, Function[]> = {};
  Object.defineProperty(window, "visualViewport", {
    value: {
      height,
      addEventListener: (event: string, fn: Function) => {
        listeners[event] = listeners[event] ?? [];
        listeners[event].push(fn);
      },
      removeEventListener: (event: string, fn: Function) => {
        listeners[event] = (listeners[event] ?? []).filter((f) => f !== fn);
      },
      // Helper to trigger resize
      _triggerResize: (newHeight: number) => {
        (window.visualViewport as { height: number }).height = newHeight;
        for (const fn of listeners.resize ?? []) {
          fn();
        }
      },
    },
    writable: true,
    configurable: true,
  });
}

**Step B: Rewrite `fab.tsx` (GREEN phase)**

Replace the entire `fab.tsx` with the keyboard-aware, route-filtered version per D-07 and D-08.

```typescript
"use client";

import { Button } from "@dashboard-leads-profills/ui/components/button";
import { Plus } from "lucide-react";
import Link from "next/link";
import { usePathname } from "next/navigation";
import { useEffect, useState } from "react";

const LEADS_NEW_HREF = "/leads/new" as unknown as "/";
const VISIBLE_ROUTES = ["/leads", "/dashboard"];

export function useKeyboardVisible(): boolean {
  const [isKeyboardVisible, setIsKeyboardVisible] = useState(false);

  useEffect(() => {
    const viewport = window.visualViewport;
    if (!viewport) {
      return;
    }

    const threshold = 0.75;

    function handleResize() {
      if (!viewport) return;
      const keyboardOpen = viewport.height < window.innerHeight * threshold;
      setIsKeyboardVisible(keyboardOpen);
    }

    viewport.addEventListener("resize", handleResize);
    return () => viewport.removeEventListener("resize", handleResize);
  }, []);

  return isKeyboardVisible;
}

export default function FAB() {
  const pathname = usePathname();
  const isKeyboardVisible = useKeyboardVisible();

  const isVisibleRoute = VISIBLE_ROUTES.some((route) => pathname === route);

  if (!isVisibleRoute || isKeyboardVisible) {
    return null;
  }

  return (
    <Link href={LEADS_NEW_HREF}>
      <Button
        aria-label="Adicionar novo lead"
        className="fixed right-6 bottom-6 z-50 h-14 w-14 rounded-full shadow-lg"
        size="icon-lg"
      >
        <Plus className="size-6" />
      </Button>
    </Link>
  );
}
```

1. Added `useKeyboardVisible` hook (exported for testing)

4. `VISIBLE_ROUTES` uses exact match (`===`) -- `/leads/new` and `/leads/[id]` will NOT match
5. FAB position unchanged: `fixed right-6 bottom-6 z-50 h-14 w-14` (bottom-right, no overlap with sidebar trigger which is top-left)

**Step A: Convert `lead-form.tsx` to CSS grid responsive per D-09**

1. **Widen the Card max-width for 2-column desktop:**
   Change `<Card className="mx-auto mb-8 w-full max-w-[480px]">` to `<Card className="mx-auto mb-8 w-full max-w-none md:max-w-2xl">`.

   Also change the header: `<header className="mx-auto flex w-full max-w-[480px] items-center gap-4 pb-6">` to `<header className="mx-auto flex w-full max-w-none items-center gap-4 pb-6 md:max-w-2xl">`.

   Rationale (Pitfall #5 in RESEARCH): 480px is too narrow for a 2-column grid on desktop. `max-w-2xl` = 672px which accommodates 2 columns comfortably. On mobile (< 480px), `max-w-none` removes the constraint entirely (screen is already narrower).
