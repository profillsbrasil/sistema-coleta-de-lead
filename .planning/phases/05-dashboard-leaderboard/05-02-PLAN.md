    **LeaderboardTab** (`apps/web/src/app/dashboard/leaderboard-tab.tsx`) per D-05, D-06, D-07, D-08, D-09, D-10:

    Client component that fetches leaderboard via tRPC, caches in Dexie, reads cache when offline.

    ```typescript
    "use client";

    import { useEffect, useState } from "react";
    import { useLiveQuery } from "dexie-react-hooks";
    import { useQuery } from "@tanstack/react-query";

    import { Skeleton } from "@dashboard-leads-profills/ui/components/skeleton";
    import { LeaderboardEntry } from "@/components/leaderboard-entry";
    import { StalenessIndicator } from "@/components/staleness-indicator";
    import { db } from "@/lib/db/index";
    import { trpc } from "@/utils/trpc";

    export default function LeaderboardTab({ userId }: { userId: string }) {
      const [lastSyncAt, setLastSyncAt] = useState<string | null>(null);

      // Fetch from server via tRPC
      const { data: serverData, error, isLoading } = useQuery(
        trpc.leaderboard.getRanking.queryOptions()
      );

      // Read cached data from Dexie
      const cachedEntries = useLiveQuery(
        () => db.leaderboardCache.orderBy("rank").toArray(),
        []
      );

      // When server data arrives, cache it in Dexie
      useEffect(() => {
        if (!serverData) return;

        const cacheData = async () => {
          await db.leaderboardCache.clear();
          const entries = serverData.ranking.map((r, i) => ({
            userId: r.userId,
            name: r.name,
            totalLeads: r.totalLeads,
            score: r.score,
            rank: i + 1,
            lastSyncAt: serverData.serverTimestamp,
          }));
          await db.leaderboardCache.bulkPut(entries);
          setLastSyncAt(serverData.serverTimestamp);
        };

        cacheData();
      }, [serverData]);

      // Load lastSyncAt from cached entries on mount
      useEffect(() => {
        if (cachedEntries && cachedEntries.length > 0 && !lastSyncAt) {
          setLastSyncAt(cachedEntries[0].lastSyncAt);
        }
      }, [cachedEntries, lastSyncAt]);

      // Refresh staleness indicator every 60 seconds
      const [, setTick] = useState(0);
      useEffect(() => {
        const interval = setInterval(() => setTick((t) => t + 1), 60_000);
        return () => clearInterval(interval);
      }, []);

      // Determine which data to display
      const displayEntries = serverData
        ? serverData.ranking.map((r, i) => ({ ...r, rank: i + 1, lastSyncAt: serverData.serverTimestamp }))
        : cachedEntries ?? [];

      // Loading state: first fetch with no cache
      if (isLoading && (!cachedEntries || cachedEntries.length === 0)) {
        return (
          <div className="space-y-2" aria-busy="true">
            <Skeleton className="h-4 w-40" />
            <Skeleton className="h-20" />
            <Skeleton className="h-20" />
            <Skeleton className="h-20" />
          </div>
        );
      }

      // Empty state: no cache and offline or error
      if (displayEntries.length === 0) {
        return (
          <p className="py-8 text-center text-sm text-muted-foreground">
            Conecte-se a internet para ver o ranking da equipe.
          </p>
        );
      }

      return (
        <div className="space-y-2">
          <StalenessIndicator lastSyncAt={lastSyncAt} />
          <ol className="space-y-2">
            {displayEntries.map((entry) => (
              <li key={entry.userId}>
                <LeaderboardEntry
                  rank={entry.rank}
                  name={entry.name}
                  totalLeads={entry.totalLeads}
                  score={entry.score}
                  isCurrentUser={entry.userId === userId}
                />
              </li>
            ))}
          </ol>
        </div>
      );
    }
    ```

    Key constraints:
    - Server fetch via tRPC React Query (per D-08)
    - Cache write to Dexie leaderboardCache on success (per D-09)
    - Read from Dexie cache via useLiveQuery when offline (per D-06, D-09)
    - StalenessIndicator with timestamp (per D-10)
    - 60s setInterval for staleness refresh (per Open Question 2 from RESEARCH.md)
    - `<ol>` semantic list with `<li>` entries (per Accessibility)
    - isCurrentUser via userId comparison (per D-06)
    - Empty state copy: "Conecte-se a internet para ver o ranking da equipe." (per Copywriting Contract)
    - Loading: 3 Skeleton entries (per UI-SPEC States)
    - Error: if server fails but cache exists, show cache with staleness (toast handled by tRPC error config)
