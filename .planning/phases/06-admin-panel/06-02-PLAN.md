2. **apps/web/src/app/admin/layout.tsx** — Server Component. Auth guard pattern from dashboard page.tsx:

```typescript
import { redirect } from "next/navigation";
import { createClient } from "@/lib/supabase/server";
import { SidebarProvider, SidebarTrigger } from "@dashboard-leads-profills/ui/components/sidebar";
import AdminSidebar from "@/components/admin-sidebar";

export default async function AdminLayout({ children }: { children: React.ReactNode }) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) redirect("/login");

  // Check admin role via JWT claims
  const { data } = await supabase.auth.getSession();
  const claims = data?.session?.user?.app_metadata as Record<string, unknown> | null;
  const userRole = claims?.user_role ?? (data?.session?.user?.user_metadata as Record<string, unknown>)?.user_role;
  if (userRole !== "admin") redirect("/dashboard");

  return (
    <SidebarProvider>
      <div className="flex min-h-[calc(100svh-49px)] w-full">
        <AdminSidebar />
        <main className="flex-1 p-6">
          <div className="mb-4 md:hidden">
            <SidebarTrigger />
          </div>
          {children}
        </main>
      </div>
    </SidebarProvider>
  );
}
```

Note: The exact method to read `user_role` from claims depends on how it's stored. Check how `packages/api/src/context.ts` reads `user_role` from getClaims() and replicate. If getClaims() is available on server client, use that. Otherwise check `app_metadata.user_role` or query `user_roles` table as fallback. The key is: if role is not "admin", redirect to /dashboard.
